//Bu kodla RFID kartlarýnýn UID numaralarý sisteme kaydolunur.
#include &lt;SPI.h&gt;
#include &lt;MFRC522.h&gt;
#include &lt;EEPROM.h&gt;
#define RST_PIN 9
#define SS_PIN 10
byte readCard[4];
int successRead;

MFRC522 mfrc522(SS_PIN, RST_PIN);
MFRC522::MIFARE_Key key;

void setup()
{
Serial.begin(9600);
SPI.begin();}

void loop()
{
mfrc522.PCD_Init();
Serial.println(&quot;Kartinizi okutunuz lutfen&quot;);
Serial.println(&quot;--------------------------&quot;);
Serial.println();
do {
//okuma baþarýlý olana kadar getID fonksiyonunu çaðýr
successRead = getID();
}



while (!successRead);
for ( int i = 0; i &lt; mfrc522.uid.size; i++ ){
//kartýn UID&#39;sini EEPROM&#39;a kaydet
EEPROM.write(i, readCard[i] );
}
delay(3000);
}

int getID() {
//yeni bir kart okunmadýysa 0 döndür
if ( ! mfrc522.PICC_IsNewCardPresent()) {
return 0;
}
if ( ! mfrc522.PICC_ReadCardSerial()) {
return 0;
} //kartýn UID&#39;sini byte byte oku ve seri monitöre yaz
for (int i = 0; i &lt; mfrc522.uid.size; i++) { //
readCard[i] = mfrc522.uid.uidByte[i];
Serial.print(readCard[i], HEX);
}
Serial.println(&quot;&quot;);
//kart okumayý durdur ve 1 döndür (okuma baþarýlý)
mfrc522.PICC_HaltA();
return 1;
}
